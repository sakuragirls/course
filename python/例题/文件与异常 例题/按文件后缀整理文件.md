```
import os
dir='E://web_clipper_chrome_2'
for file in os.listdir(dir): #遍历dir中所有的文件
    ext=os.path.splitext(file)[-1]#获取每一个文件的后缀
    print(file,ext) #打印所有文件及其后缀
```
